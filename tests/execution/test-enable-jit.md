# test-enable-jit

```graphql @config
schema
  @server(port: 8000, hostname: "0.0.0.0", enableJIT: true)
  @upstream(baseURL: "http://jsonplaceholder.typicode.com") {
  query: Query
}

type Query @cache(maxAge: 30000) {
  posts: [Post] @http(path: "/posts")
}

type User {
  id: Int!
  name: String!
  username: String!
}

type Post {
  id: Int!
  userId: Int!
  title: String!
  user: User @http(path: "/users/{{.value.userId}}")
}
```
