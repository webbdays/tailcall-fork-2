  query: Query

  id: Int!
  userId: Int!
  user: User @graphQL(args: [{key: "id", value: "{{.value.userId}}"}], name: "user")
}

type Query {
  post(id: Int!): Post @http(baseURL: "http://jsonplaceholder.typicode.com", path: "/posts/{{.args.id}}")
}

type User {
  id: Int
  name: String
}
